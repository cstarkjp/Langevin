51bool SimDP::initialize(int n_decimals)
52{

54 std::cout

60 std::cout
61 << "SimDP::initialize failure: couldn't initialize grid"
62 << std::endl;
63 return false;
64 }
65 dpLangevin->prepare(coefficients);
66 this->n_decimals = n_decimals;
67 n_epochs = count_epochs();
68 t_epochs = dbl_vec_t(n_epochs, 0.0);
69 mean_densities = dbl_vec_t(n_epochs, 0.0);
70 // Treat epoch#0 as the initial grid state
71 // So after initialization, we are nominally at epoch#1
72 i_next_epoch = 1;
73 t_next_epoch = p.dt;
74 if (not choose_integrator())
75 {
76 std::cout << "SimDP::initialize failure: unable to choose integrator" << std::endl;
77 return false;
78 }
79 if (not dpLangevin->check_boundary_conditions(p))
80 {
81 std::cout << "SimDP::initialize failure: wrong number of boundary conditions" << std::endl;
82 return false;
83 }
