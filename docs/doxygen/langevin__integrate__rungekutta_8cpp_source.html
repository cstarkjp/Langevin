12// Parallelizing this is going to be complicated by the fact that each
13// call to ddensitydt_nonlinear() entails reference to the neighboring cells
14// of density_grid[i], which gets fiddly at the grid boundaries.
15// Don't know how all this will map onto a GPU treatment.
16void BaseLangevin::integrate_rungekutta(rng_t& rng)
17{
18 // Runge-Kutta 1st step
19 for (auto i=0; i<n_cells; i++)
20 {
21 // k1 = dρ/dt at start-interval t
22 k1_grid[i] = ddensitydt_nonlinear(i, density_grid)*dt;
23 density_plusk1_grid[i] = density_grid[i] + k1_grid[i]/2;
24 }
25
26 // Runge-Kutta 2nd step
27 for (auto i=0; i<n_cells; i++)
28 {
29 // k2 = dρ/dt at mid-interval t+Δt/2 using k1
30 k2_grid[i] = ddensitydt_nonlinear(i, density_plusk1_grid)*dt;
31 density_plusk2_grid[i] = density_grid[i] + k2_grid[i]/2;
32 }
33
34 // Runge-Kutta 3rd step
35 for (auto i=0; i<n_cells; i++)
36 {
37 // k3 = dρ/dt at mid-interval t+Δt/2 using k2
38 k3_grid[i] = ddensitydt_nonlinear(i, density_plusk2_grid)*dt;
39 density_plusk3_grid[i] = density_grid[i] + k3_grid[i];
40 }
41
42 // Runge-Kutta 4th step
43 for (auto i=0; i<n_cells; i++)
44 {
45 // k4 = dρ/dt at end-interval t+Δt using k3
46 k4_grid[i] = ddensitydt_nonlinear(i, density_plusk3_grid)*dt;
47 // Estimate for ρ(t+Δt) = ρ(t) + (k1 + 2*k2 + 2*k3 + k4)/6
48 density_grid[i] += (k1_grid[i]+2*(k2_grid[i]+k3_grid[i])+k4_grid[i])/6;
49 }
50
51 // Stochastic step - strictly local to each i.
52 // This step combines the linear and noise terms in the Langevin equation.
53 mean_density = 0.0;
54 for (auto i=0; i<n_cells; i++)
55 {
56 // Generate a random sample from the FPE-derived noise PDF
57 // whose form is a kind of modified Bessel function,
58 // which is also a Poisson-randomized gamma distribution.
59 // Do so by sampling: (1) a Poisson variate; (2) a gamma variate
60 // whose shape parameter is that Poisson sample (an integer).
61 poisson_sampler = poisson_dist_t(lambda_scaled*density_grid[i]);
62 gamma_sampler = gamma_dist_t(poisson_sampler(rng), 1/lambda);
63 // Update density value at this grid location i
64 density_grid[i] = gamma_sampler(rng);
65 // Incrementally compute mean density
66 mean_density += density_grid[i];
67 }
68 mean_density /= static_cast<double>(n_cells);
69}
